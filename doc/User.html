def movie_list
  seen_movies = Array.new
  movie_ratings.each {|movie|
    seen_movies.push(movie[:movie_id])
  }
  return seen_movies
end

def rating(m)
  found_movie = movie_ratings.find {|movie| movie[:movie_id] == m}
  if found_movie.nil?
    return 0
  else
    return found_movie[:rating]
  end
end
